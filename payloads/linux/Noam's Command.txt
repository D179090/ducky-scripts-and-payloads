REM     Author : benur
REM     --> This script adds an alias to the .bashrc file and launches it


REM     ###     START OF SCRIPT
DELAY 1000


REM     ###     Opening the terminal
GUI
DELAY 450
STRING terminal
DELAY 200
ENTER
DELAY 750


REM     ###     Appending the alias
STRING echo "alias <NAME OF ALIAS>='<YOUR COMMAND HERE'" >> ~/.bashrc
DELAY 200
ENTER
DELAY 450


REM     ###     Recompiling .bashrc
STRING source .bashrc
DELAY 200
ENTER
DELAY 450


REM     ###     Launching the new alias
STRING <NAME OF ALIAS>
DELAY 200
ENTER

REM     ###     END OF SCRIPT
